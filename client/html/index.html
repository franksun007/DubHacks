	"get_next_moves": function(germs) {
		var command = [];
		for (var i = 0; i < germs.length; i++) {
			var direction = ["left", "right", "up", "down"];
	    	var action = ["move", "split"];
			var actionindex = parseInt(Math.random() * 2);

	}
